```python
"""Behavior contract — tool boundary declarations and AST compliance verification.

Every tool generated by the autonomous evolution loop must declare a
BehaviorContract specifying its input/output schema, allowed side effects,
and maximum runtime. The contract is verified via AST analysis before the
tool enters the sandbox forge pipeline.
"""

from __future__ import annotations
from dataclasses import dataclass, field
import ast

# Modules whose import implies specific side effects.
# When side_effects=["none"], NONE of these modules may be imported.
_MODULE_SIDE_EFFECT_MAP: dict[str, str] = {
    # Network
    "urllib": "network",
    "http": "network",
    "httpx": "network",
    "requests": "network",
    "aiohttp": "network",
    "socket": "network",
    "ftplib": "network",
    "smtplib": "network",
    # File I/O
    "pathlib": "file_read",
    "os": "file_write",
    "sys": "file_write",
    "shutil": "file_write",
    "io": "file_write",
    "csv": "file_read",
    "json": "file_read",
    # Subprocess
    "subprocess": "subprocess",
    "multiprocessing": "subprocess",
    "signal": "subprocess",
}

# STDlib modules that are always allowed regardless of side_effects declaration
_ALWAYS_ALLOWED = frozenset({
    "math", "statistics", "random", "datetime", "collections",
    "itertools", "functools", "typing", "hashlib", "base64",
    "re", "string", "textwrap", "enum", "dataclasses", "uuid",
    "copy", "html", "xml", "argparse", "logging", "json",
})


class ContractValidationError(ValueError):
    """Raised when a BehaviorContract is malformed or inconsistent."""
    pass


@dataclass
class ContractComplianceResult:
    """Result of verifying a BehaviorContract against actual code."""
    compliant: bool
    violations: list[str] = field(default_factory=list)


@dataclass
class BehaviorContract:
    """Tool behavior boundary declaration.

    Produced by the LLM alongside code generation. Verified via AST
    before sandbox forging. Monitored at runtime for violations.

    Attributes:
        tool_name: Name of the tool this contract applies to.
        input_schema: JSON Schema describing expected input parameters.
        output_schema: JSON Schema describing expected return value.
        side_effects: Allowed side-effect categories. Valid values:
            "file_read", "file_write", "network", "subprocess", "none".
        max_runtime_ms: Maximum allowed execution time in milliseconds.
    """

    tool_name: str
    input_schema: dict
    output_schema: dict
    side_effects: list[str]
    max_runtime_ms: int

    _VALID_EFFECTS: set[str] = field(
        default=frozenset({"file_read", "file_write", "network", "subprocess", "none"}),
        init=False, repr=False,
    )

    @classmethod
    def from_generated(cls, tool_name: str, contract_json: dict) -> "BehaviorContract":
        """Parse a contract from LLM-generated JSON.

        Validates that side_effects only contains recognized values.
        Fills in safe defaults for missing optional fields.

        Args:
            tool_name: Name of the tool.
            contract_json: Raw dict from LLM contract output.

        Returns:
            BehaviorContract with validated side_effects and defaults.

        Raises:
            ContractValidationError: If side_effects contains invalid values.
        """
        side_effects = contract_json.get("side_effects", ["none"])
        if not isinstance(side_effects, list):
            raise ContractValidationError(
                f"side_effects must be a list, got {type(side_effects).__name__}"
            )

        invalid = set(side_effects) - cls._VALID_EFFECTS
        if invalid:
            raise ContractValidationError(
                f"Invalid side effects: {sorted(invalid)}. "
                f"Valid values: {sorted(cls._VALID_EFFECTS)}"
            )

        return cls(
            tool_name=tool_name,
            input_schema=contract_json.get("input_schema", {}),
            output_schema=contract_json.get("output_schema", {}),
            side_effects=side_effects,
            max_runtime_ms=contract_json.get("max_runtime_ms", 5000),
        )

    def verify_code_compliance(self, code: str) -> ContractComplianceResult:
        """AST-parse the code and verify its imports match declared side_effects.

        Checks every import statement against the declared side_effects.
        If side_effects=["none"], any module in _MODULE_SIDE_EFFECT_MAP
        is a violation. If side_effects includes "network", then urllib/
        requests/etc. are allowed.

        Modules in _ALWAYS_ALLOWED (math, json, datetime, etc.) are never
        flagged regardless of side_effects declaration.

        Args:
            code: Python source code to analyze.

        Returns:
            ContractComplianceResult with compliant flag and violation list.
        """
        try:
            tree = ast.parse(code)
        except SyntaxError as e:
            return ContractComplianceResult(
                compliant=False,
                violations=[f"Syntax error in code: {e}"],
            )

        violations: list[str] = []
        declared = set(self.side_effects)

        for node in ast.walk(tree):
            imported_modules: list[str] = []

            if isinstance(node, ast.Import):
                for alias in node.names:
                    imported_modules.append(alias.name.split(".")[0])
            elif isinstance(node, ast.ImportFrom):
                if node.module:
                    imported_modules.append(node.module.split(".")[0])

            for mod in imported_modules:
                if mod in _ALWAYS_ALLOWED:
                    continue
                expected_effect = _MODULE_SIDE_EFFECT_MAP.get(mod)
                if expected_effect is None:
                    # Unknown module — not in our map.
                    # In strict mode, flag it so sandbox can check.
                    # The sandbox allowlist is the final arbiter.
                    continue
                if expected_effect not in declared:
                    violations.append(
                        f"Module '{mod}' implies side_effect '{expected_effect}', "
                        f"but contract declares {sorted(declared)}"
                    )

        return ContractComplianceResult(
            compliant=len(violations) == 0,
            violations=violations,
        )
```

"""Autonomous Evolution Loop — closed 8-stage evolution cycle.

Replaces ImprovementLoop. Full zero-human-intervention cycle:
  GAP_DETECT → SPEC_DESIGN → CODE_GENERATE → CONTRACT_CHECK
    → SANDBOX_FORGE → REGISTER → ONLINE_VERIFY → EVALUATE

Safety: sandbox (AST allowlist) → behavior contract (import checking)
    → online verify (call testing) → quality delta (rollback guard).
"""

from __future__ import annotations
import json
import logging
import re
import threading
import time as time_module

# ── Result types ─────────────────────────────────────────────────────────

@dataclass
class CycleResult:
    """Result of a single evolution cycle."""
    success: bool
    skipped: bool = False
    stage: str = ""
    spec: ImprovementSpec | None = None
    code: str | None = None
    contract: BehaviorContract | None = None
    error: str = ""
    details: dict = field(default_factory=dict)

    @classmethod
    def skipped_result(cls, assessment) -> "CycleResult":
        return cls(success=False, skipped=True, stage="skipped",
                   details={"reason": "No improvement need detected"})

    @classmethod
    def failed(cls, stage: str, spec: ImprovementSpec | None = None,
               error: str = "", **details) -> "CycleResult":
        return cls(success=False, stage=stage, spec=spec,
                   error=error, details=details)

    @classmethod
    def success_result(cls, spec: ImprovementSpec, code: str,
                       contract: BehaviorContract) -> "CycleResult":
        return cls(success=True, stage="evaluate",
                   spec=spec, code=code, contract=contract)


@dataclass
class ToolSnapshot:
    """State captured before a tool deployment — used for rollback."""
    tool_name: str
    code: str | None
    tool_list_snapshot: dict
    forged_list_snapshot: dict
    captured_at: str


@dataclass
class QualityDelta:
    """Comparison between pre-deployment and post-deployment quality."""
    degraded: bool
    reason: str = ""


# ── Main Loop ────────────────────────────────────────────────────────────

class AutonomousEvolutionLoop:
    """Closed autonomous evolution — gap to deployment, zero human touch.

    Runs as a background thread, independent of the PRAL cognitive loop.
    Reads tool/knowledge state via Plugins, writes new tools via ToolPlugin.

    Constructor Args:
        llm_backend: The agent's LLM backend. Must have create_message().
        tool_plugin: ToolPlugin instance from kernel lifecycle.
        knowledge_plugin: KnowledgePlugin instance from kernel lifecycle.
        lineage: LineageTracker for recording evolution events.
    """

    def __init__(
        self,
        llm_backend: Any,
        tool_plugin: Any,
        knowledge_plugin: Any,
        lineage: Any,
    ):
        self._llm = llm_backend
        self._tools = tool_plugin
        self._knowledge = knowledge_plugin
        self._lineage = lineage

        # ── Configuration (overridable from config.yaml) ──
        self.min_interval_seconds: int = 300
        self.max_improvements_per_session: int = 10
        self.max_generate_retries: int = 3
        self.rollback_on_failure_count: int = 3
        self.contract_enforcement: str = "strict"  # strict | warn | off

        # ── Runtime state ──
        self._running: bool = False
        self._paused: bool = False
        self._improvements_this_session: int = 0
        self._last_cycle_at: Optional[str] = None
        self._cycle_history: list[dict] = []
        self._thread: Optional[threading.Thread] = None
        self._snapshots: dict[str, ToolSnapshot] = {}

        # ── Trigger config (same dimensions as old ImprovementLoop) ──
        self.trigger_config: dict = {
            "min_trigger_score": 0.01,
            "capability_gap":   {"enabled": True, "threshold": 0.0,  "weight": 0.08},
            "code_health":      {"enabled": True, "threshold": 0.50, "weight": 0.20},
            "knowledge_fresh":  {"enabled": True, "threshold": 0.30, "weight": 0.20},
            "tool_fitness":     {"enabled": True, "threshold": 0.10, "weight": 0.12},
            "tool_dedup":       {"enabled": True, "threshold": 0.40, "weight": 0.08},
            "subgraph_balance": {"enabled": True, "threshold": 0.30, "weight": 0.12},
            "task_completion":  {"enabled": True, "threshold": 0.30, "weight": 0.15},
            "goal_achievement": {"enabled": True, "threshold": 0.30, "weight": 0.10},
        }
        self._last_trigger_scores: dict = {}
        self._last_triggered_by: list = []

        self._load_state()

    # ── Public API ────────────────────────────────────────────────────────

    def configure(self, **kwargs) -> None:
        """Apply configuration overrides from config.yaml evolution section."""
        for key in ("min_interval_seconds", "max_improvements_per_session",
                     "max_generate_retries", "rollback_on_failure_count",
                     "contract_enforcement"):
            if key in kwargs and kwargs[key] is not None:
                setattr(self, key, kwargs[key])

    def start(self) -> dict:
        """Start the evolution loop (background thread)."""
        if self._running:
            return {"success": False, "error": "Loop is already active."}

        self._running = True
        self._paused = False
        self._improvements_this_session = 0

        self._thread = threading.Thread(
            target=self._run_loop, daemon=True, name="evolution-loop",
        )
        self._thread.start()
        logger.info("Autonomous evolution loop started (background thread)")
        return {"success": True, "message": "Evolution loop started."}

    def stop(self) -> dict:
        """Stop the evolution loop."""
        self._running = False
        self._save_state()
        logger.info("Evolution loop stopped.")
        return {"success": True, "message": "Evolution loop stopped."}

    def pause(self) -> dict:
        """Pause the evolution loop (preserves state)."""
        self._paused = True
        return {"success": True, "message": "Loop paused."}

    def resume(self) -> dict:
        """Resume a paused evolution loop."""
        self._paused = False
        return {"success": True, "message": "Loop resumed."}

    def run_one_cycle(self) -> CycleResult:
        """Execute one full evolution cycle synchronously."""
        # 1. GAP_DETECT
        assessment = self._assess_need()
        if not assessment.get("should_trigger", False):
            return CycleResult.skipped_result(assessment)

        # 2. SPEC_DESIGN
        spec = self._generate_spec(assessment)
        if spec is None:
            return CycleResult.failed("spec_design",
                                      error="Could not generate improvement spec")

        # 3. CODE_GENERATE — LLM generates code + behavior contract
        try:
            code, contract = self._generate_code(spec)
        except Exception as e:
            logger.exception("Code generation failed")
            return CycleResult.failed("code_generation", spec, error=str(e))

        if code is None:
            return CycleResult.failed("code_generation", spec,
                                      error="LLM returned no code")

        # 4. CONTRACT_CHECK
        contract_ok = self._check_contract(code, contract)
        if not contract_ok:
            if self.contract_enforcement == "strict":
                for attempt in range(self.max_generate_retries - 1):
                    logger.info("Contract check failed, retry %d/%d",
                                attempt + 2, self.max_generate_retries)
                    try:
                        code, contract = self._generate_code(spec, retry=True)
                    except Exception:
                        continue
                    if self._check_contract(code, contract):
                        break
                else:
                    return CycleResult.failed("contract_check", spec,
                                              error="Contract check failed after retries")

        # 5. SANDBOX_FORGE
        forge_result = self._tools.forge_cycle(spec, code=code)
        if not forge_result.success:
            return CycleResult.failed("sandbox_forge", spec,
                                      error="Sandbox forge rejected the code",
                                      forge_stages=[s.to_dict() for s in forge_result.stages])

        # 6. REGISTER — already done by forge_cycle; confirm
        tool_exists = spec.function_name in self._tools.list_tools()
        if not tool_exists:
            return CycleResult.failed("register", spec,
                                      error="Tool not found in registry after forge")

        # 7. ONLINE_VERIFY
        snapshot_before = self._capture_snapshot(spec.function_name)
        verify_result = self._verify_online(spec.function_name)
        if verify_result.consecutive_failures >= self.rollback_on_failure_count:
            self._rollback(spec.function_name, snapshot_before)
            return CycleResult.failed("online_verify", spec,
                                      error=f"Consecutive failures: {verify_result.consecutive_failures}")

        # 8. EVALUATE
        quality_delta = self._evaluate_quality_delta(snapshot_before)
        if quality_delta.degraded:
            self._rollback(spec.function_name, snapshot_before)
            return CycleResult.failed("evaluate", spec,
                                      error=f"Quality degraded: {quality_delta.reason}")

        # Success
        self._improvements_this_session += 1
        self._save_snapshot(spec.function_name, code)
        if self._lineage:
            try:
                self._lineage.record_evolution(spec, contract)
            except Exception:
                logger.debug("Lineage recording skipped — tracker unavailable")

        self._last_cycle_at = now().isoformat()
        self._cycle_history.append({
            "timestamp": self._last_cycle_at,
            "tool_name": spec.function_name,
            "success": True,
        })
        self._save_state()

        logger.info("Evolution cycle complete: %s", spec.function_name)
        return CycleResult.success_result(spec, code, contract)

    def export_state(self) -> dict:
        """Return loop state for evolution_metrics.py compatibility."""
        return {
            "running": self._running,
            "paused": self._paused,
            "improvements_this_session": self._improvements_this_session,
            "last_cycle_at": self._last_cycle_at,
            "cycle_count": len(self._cycle_history),
            "trigger_scores": self._last_trigger_scores,
            "triggered_by": self._last_triggered_by,
        }

    def execute_once_if_needed(self) -> dict:
        """Compatibility shim for cognitive_loop.py's existing call pattern.

        Returns a dict compatible with what CognitiveLoop.learn() expects.
        """
        assessment = self._assess_need()
        if not assessment.get("should_trigger", False):
            return {"triggered": False, "assessment": assessment}

        if self._paused:
            return {"triggered": False, "reason": "paused"}

        if self._improvements_this_session >= self.max_improvements_per_session:
            return {"triggered": False, "reason": "quota_exhausted"}

        result = self.run_one_cycle()
        return {
            "triggered": True,
            "assessment": assessment,
            "result": {
                "success": result.success,
                "summary": f"Stage: {result.stage}, "
                           f"Tool: {result.spec.function_name if result.spec else 'N/A'}",
            },
        }

    # ── Stage 1: Gap Detection ────────────────────────────────────────────

    def _assess_need(self) -> dict:
        """Multi-dimensional improvement need assessment.

        Same 8 dimensions as old ImprovementLoop. Each dimension evaluator
        gracefully degrades to 0.0 when the corresponding forged tool is
        unavailable (first session, no tools forged yet).
        """
        scores = {}
        for dim_name, config in self.trigger_config.items():
            if dim_name == "min_trigger_score" or not config.get("enabled", True):
                continue
            score = self._evaluate_dimension(dim_name, config.get("threshold", 0))
            scores[dim_name] = score

        total_weight = sum(
            self.trigger_config.get(d, {}).get("weight", 0)
            for d in scores
            if self.trigger_config.get(d, {}).get("enabled", True)
        )

        need_score = sum(
            scores[d] * self.trigger_config.get(d, {}).get("weight", 0) / total_weight
            for d in scores
        ) if total_weight > 0 else 0

        triggered_by = [
            {"dimension": d, "score": scores[d],
             "threshold": self.trigger_config[d]["threshold"]}
            for d in scores
            if scores[d] > 0
        ]

        should_trigger = need_score >= self.trigger_config["min_trigger_score"]
        self._last_trigger_scores = scores
        self._last_triggered_by = triggered_by

        return {
            "need_score": need_score,
            "min_trigger_score": self.trigger_config["min_trigger_score"],
            "scores": scores,
            "triggered_by": triggered_by,
            "should_trigger": should_trigger,
        }

    def _evaluate_dimension(self, dim_name: str, threshold: float) -> float:
        """Evaluate a single improvement dimension. Returns 0.0–1.0.

        All evaluators degrade gracefully: if the required forged tool
        is unavailable, returns 0.0 (no false triggers).
        """
        evaluators = {
            "capability_gap": self._eval_capability_gap,
            "code_health": self._eval_code_health,
            "knowledge_fresh": self._eval_knowledge_fresh,
            "tool_fitness": self._eval_tool_fitness,
            "tool_dedup": self._eval_tool_dedup,
            "subgraph_balance": self._eval_subgraph_balance,
            "task_completion": self._eval_task_completion,
            "goal_achievement": self._eval_goal_achievement,
        }
        evaluator = evaluators.get(dim_name)
        if not evaluator:
            return 0.0
        try:
            score = evaluator(threshold)
        except Exception:
            logger.debug("Dimension evaluator '%s' failed — returning 0.0", dim_name)
            return 0.0
        return max(0.0, min(1.0, score))

    def _eval_capability_gap(self, threshold: float) -> float:
        """Evaluate uncovered capability gaps.

        Simplified: counts tools vs a baseline. Returns 0 if at least
        5 tools exist (basic coverage), otherwise a proportional gap score.
        """
        tools = self._tools.list_tools()
        count = len(tools)
        if count >= 10:
            return 0.0
        return max(0.0, (10 - count) / 10.0)

    def _eval_code_health(self, threshold: float) -> float:
        """Evaluate code health via forged code_entropy tool."""
        try:
            import tain_agent.tools.forged.code_entropy as ce
            result = ce.analyze_entropy()
            health_score = result.get("health_score", 1.0)
            return max(0.0, threshold - health_score) if health_score < threshold else 0.0
        except (ImportError, Exception):
            return 0.0

    def _eval_knowledge_fresh(self, threshold: float) -> float:
        """Evaluate knowledge freshness via forged knowledge_freshness tool."""
        try:
            import tain_agent.tools.forged.knowledge_freshness as kf
            result = kf.check_freshness()
            fresh_ratio = result.get("fresh_ratio", 1.0)
            if fresh_ratio == 0.0:
                return 1.0
            return max(0.0, threshold - fresh_ratio) if fresh_ratio < threshold else 0.0
        except (ImportError, Exception):
            return 0.0

    def _eval_tool_fitness(self, threshold: float) -> float:
        """Evaluate dead tool ratio via forged tool_fitness tool."""
        try:
            import tain_agent.tools.forged.tool_fitness as tf
            result = tf.analyze_fitness()
            dead_ratio = result.get("dead_tool_ratio", 0.0)
            return dead_ratio if dead_ratio > threshold else 0.0
        except (ImportError, Exception):
            return 0.0

    def _eval_tool_dedup(self, threshold: float) -> float:
        """Detect duplicate tool implementations by content hash."""
        try:
            import hashlib
            tools = self._tools.list_forged()
            if not tools or len(tools) < 2:
                return 0.0
            hashes = {}
            total = 0
            duplicates = 0
            for name, info in tools.items():
                total += 1
                code = info.get("code", "") if isinstance(info, dict) else str(info)
                h = hashlib.md5(code.encode()).hexdigest()
                if h in hashes:
                    duplicates += 1
                else:
                    hashes[h] = name
            if total == 0:
                return 0.0
            dup_ratio = duplicates / total
            return dup_ratio if dup_ratio > threshold else 0.0
        except Exception:
            return 0.0

    def _eval_subgraph_balance(self, threshold: float) -> float:
        """Evaluate knowledge subgraph balance via forged tool."""
        try:
            import tain_agent.tools.forged.knowledge_subgraph as ks
            result = ks.check_balance()
            balance_score = result.get("balance_score", 1.0)
            return max(0.0, threshold - balance_score) if balance_score < threshold else 0.0
        except (ImportError, Exception):
            return 0.0

    def _eval_task_completion(self, threshold: float) -> float:
        """Stub — task completion evaluation requires decision_log data.

        Not wired in v0.10.0; returns 0.0 to avoid false triggers.
        Full implementation needs access to decision_log from cognitive_loop.
        """
        return 0.0

    def _eval_goal_achievement(self, threshold: float) -> float:
        """Evaluate goal achievement via KnowledgePlugin's GoalManager."""
        try:
            goals = None
            if hasattr(self._knowledge, 'goals'):
                goals = self._knowledge.goals
            if goals is None:
                return 0.0
            all_goals = []
            if hasattr(goals, 'list_active'):
                all_goals.extend(goals.list_active())
            if hasattr(goals, 'list_completed'):
                all_goals.extend(goals.list_completed())
            if not all_goals:
                return 0.0
            abandoned = sum(1 for g in all_goals
                          if isinstance(g, dict) and g.get("status") in ("abandoned", "blocked"))
            return round(abandoned / len(all_goals), 4) if all_goals else 0.0
        except Exception:
            return 0.0

    # ── Stage 2: Spec Generation ──────────────────────────────────────────

    def _generate_spec(self, assessment: dict) -> ImprovementSpec | None:
        """Generate an ImprovementSpec from the need assessment.

        Uses the triggered dimension to create a targeted spec.
        When triggered by capability_gap, uses the tool list to
        identify missing capability areas.
        """
        triggered_by = assessment.get("triggered_by", [])
        if not triggered_by:
            return None

        dim_name = triggered_by[0].get("dimension", "capability_gap")
        existing_tools = self._tools.list_tools()
        existing_names = set(existing_tools.keys())

        # Generate a unique function name based on the gap
        base_name = f"auto_{dim_name}"
        func_name = base_name
        counter = 1
        while func_name in existing_names:
            func_name = f"{base_name}_{counter}"
            counter += 1

        return ImprovementSpec(
            capability_id=f"evolve_{func_name}",
            description=f"Auto-generated tool to address {dim_name} gap "
                        f"(score: {triggered_by[0].get('score', 0):.3f})",
            function_name=func_name,
            parameters={},
            reasoning=f"Triggered by {dim_name} dimension",
        )

    # ── Stage 3: Code Generation ──────────────────────────────────────────

    CODE_GENERATION_SYSTEM_PROMPT = (
        "You are a Python code generator for an AI agent framework. "
        "Write clean, safe, well-documented Python functions. "
        "Every function must return a dictionary with a 'success' or 'result' key. "
        "Follow the tool specification exactly."
    )

    def _generate_code(self, spec: ImprovementSpec, retry: bool = False) -> tuple[str | None, BehaviorContract]:
        """Call LLM to generate code and behavior contract.

        Args:
            spec: The improvement specification.
            retry: If True, the prompt includes previous failure context.

        Returns:
            Tuple of (code_string, BehaviorContract). code may be None on failure.
        """
        prompt = self._build_generation_prompt(spec, retry=retry)
        try:
            response = self._llm.create_message(
                system_prompt=self.CODE_GENERATION_SYSTEM_PROMPT,
                messages=[{"role": "user", "content": prompt}],
            )
            raw_text = response.text_blocks[0] if response.text_blocks else ""
        except Exception as e:
            logger.error("LLM call failed during code generation: %s", e)
            return None, BehaviorContract(
                tool_name=spec.function_name,
                input_schema={}, output_schema={},
                side_effects=["none"], max_runtime_ms=5000,
            )

        code, contract_json = self._parse_generated_response(raw_text)
        try:
            contract = BehaviorContract.from_generated(spec.function_name, contract_json)
        except ContractValidationError as e:
            logger.warning("Contract validation failed: %s", e)
            contract = BehaviorContract(
                tool_name=spec.function_name,
                input_schema={}, output_schema={},
                side_effects=["none"], max_runtime_ms=5000,
            )

        return code, contract

    def _build_generation_prompt(self, spec: ImprovementSpec, retry: bool = False) -> str:
        """Build the LLM prompt for code generation."""
        allowlist = self._tools.get_sandbox_allowlist()
        existing_tools = self._tools.list_tools()

        parts = [
            "Write a Python function implementing this tool specification:",
            "",
            f"Function name: {spec.function_name}",
            f"Description: {spec.description}",
        ]
        if spec.parameters:
            parts.append(f"Parameters: {json.dumps(spec.parameters)}")

        parts.extend([
            "",
            "Available modules (sandbox allowlist — ONLY these imports are allowed):",
            json.dumps(allowlist, indent=2),
            "",
            "Existing tools (do NOT duplicate — use these if relevant):",
            json.dumps(sorted(existing_tools.keys()), indent=2),
            "",
            "After the code, output a behavior contract in JSON format.",
            "The contract MUST declare:",
            '- input_schema: JSON Schema of expected inputs (e.g., {"type": "object", "properties": {...}})',
            '- output_schema: JSON Schema of expected outputs',
            '- side_effects: list from ["file_read", "file_write", "network", "subprocess", "none"]',
            "- max_runtime_ms: maximum allowed execution time in milliseconds (default 5000)",
            "",
            "Output format:",
            "```python",
            "<your code here>",
            "```",
            "",
            "```contract",
            "<your contract JSON here>",
            "```",
        ])

        if retry:
            parts.insert(0, "PREVIOUS ATTEMPT FAILED CONTRACT CHECK. "
                            "Ensure your code's imports match the declared side_effects.\n")

        return "\n".join(parts)

    def _parse_generated_response(self, raw_text: str) -> tuple[str | None, dict]:
        """Parse LLM response into (code, contract_json).

        Handles markdown-fenced and bare output formats.
        """
        code: str | None = None
        contract_json: dict = {}

        # Try markdown fence extraction
        code_match = re.search(r'```python\s*\n(.*?)```', raw_text, re.DOTALL)
        if code_match:
            code = code_match.group(1).strip()
        else:
            # Try bare code — everything before the contract marker
            contract_match = re.search(r'```contract\s*\n(.*?)```', raw_text, re.DOTALL)
            if contract_match:
                code = raw_text[:raw_text.find("```contract")].strip()
            else:
                # Last resort: split on last JSON-like block
                json_match = re.search(r'\{[\s\S]*"side_effects"[\s\S]*\}', raw_text)
                if json_match:
                    code = raw_text[:json_match.start()].strip()
                else:
                    code = raw_text.strip()

        # Remove markdown fence remnants from code
        if code:
            code = re.sub(r'^```\w*\s*', '', code)
            code = re.sub(r'\s*```\s*$', '', code)

        # Extract contract JSON
        contract_match = re.search(r'```contract\s*\n(.*?)```', raw_text, re.DOTALL)
        if contract_match:
            try:
                contract_json = json.loads(contract_match.group(1).strip())
            except json.JSONDecodeError:
                logger.warning("Failed to parse contract JSON from fence block")
        else:
            # Try bare JSON with side_effects key
            json_match = re.search(r'\{[\s\S]*"side_effects"[\s\S]*\}', raw_text)
            if json_match:
                try:
                    contract_json = json.loads(json_match.group(0))
                except json.JSONDecodeError:
                    logger.warning("Failed to parse bare contract JSON")

        if not code or not code.strip():
            code = None

        return code, contract_json

    # ── Stage 4: Contract Check ───────────────────────────────────────────

    def _check_contract(self, code: str, contract: BehaviorContract) -> bool:
        """Verify code complies with its declared behavior contract."""
        try:
            result = contract.verify_code_compliance(code)
            return result.compliant
        except Exception as e:
            logger.warning("Contract compliance check error: %s", e)
            return False

    # ── Stage 7: Online Verify ────────────────────────────────────────────

    def _verify_online(self, tool_name: str) -> _VerifyResult:
        """Call the newly deployed tool to verify it works.

        Attempts 3 calls with auto-generated test inputs. Tracks
        consecutive failures for rollback decisions.
        """
        failures = 0
        # Generate simple test inputs from tool parameter info
        tools = self._tools.list_tools()
        tool_info = tools.get(tool_name, {})

        for attempt in range(3):
            try:
                result = self._tools.call(tool_name)
                if isinstance(result, dict) and result.get("success") is True:
                    pass  # Success
                elif isinstance(result, dict) and "error" not in str(result).lower():
                    pass  # Non-error response
                else:
                    failures += 1
            except Exception:
                failures += 1
                logger.debug("Online verify attempt %d failed for %s",
                            attempt + 1, tool_name)

        return _VerifyResult(consecutive_failures=failures)

    # ── Stage 8: Quality Evaluation ───────────────────────────────────────

    def _capture_snapshot(self, tool_name: str = "") -> ToolSnapshot:
        """Capture current tool and knowledge state for delta comparison."""
        tools = self._tools.list_tools()
        forged = self._tools.list_forged()
        existing_code = forged.get(tool_name, None) if tool_name else None
        if isinstance(existing_code, dict):
            existing_code = existing_code.get("code", None)

        return ToolSnapshot(
            tool_name=tool_name,
            code=existing_code,
            tool_list_snapshot=dict(tools),
            forged_list_snapshot=dict(forged),
            captured_at=now().isoformat(),
        )

    def _save_snapshot(self, tool_name: str, code: str) -> None:
        """Save a success snapshot for future rollback reference."""
        self._snapshots[tool_name] = ToolSnapshot(
            tool_name=tool_name,
            code=code,
            tool_list_snapshot=self._tools.list_tools(),
            forged_list_snapshot=self._tools.list_forged(),
            captured_at=now().isoformat(),
        )

    def _evaluate_quality_delta(self, before: ToolSnapshot) -> QualityDelta:
        """Compare current tool/knowledge state vs pre-deployment snapshot.

        Degradation criteria:
        - Tool count decreased (unexpected removal)
        - Forged tools became unimportable
        - Knowledge nodes decreased significantly (>20%)
        """
        after_tools = self._tools.list_tools()
        after_forged = self._tools.list_forged()

        # Criterion 1: Tool count stability
        if len(after_tools) < len(before.tool_list_snapshot):
            return QualityDelta(
                degraded=True,
                reason=f"Tool count decreased: {len(before.tool_list_snapshot)} → {len(after_tools)}",
            )

        # Criterion 2: Knowledge stability
        before_knowledge = getattr(self._knowledge, 'node_count', None)
        if before_knowledge is not None:
            current_knowledge = getattr(self._knowledge, 'node_count', 0)
            if before_knowledge > 0 and current_knowledge < before_knowledge * 0.8:
                return QualityDelta(
                    degraded=True,
                    reason=f"Knowledge nodes decreased: {before_knowledge} → {current_knowledge}",
                )

        return QualityDelta(degraded=False)

    # ── Rollback ──────────────────────────────────────────────────────────

    def _rollback(self, tool_name: str, snapshot: ToolSnapshot) -> None:
        """Rollback a tool deployment: remove new tool, restore old if existed."""
        logger.warning("Rolling back tool: %s", tool_name)

        # Remove the newly deployed tool
        try:
            if hasattr(self._tools, 'rollback'):
                self._tools.rollback(tool_name)
            else:
                # Fallback: remove from forge
                if hasattr(self._tools, '_forge') and self._tools._forge:
                    self._tools._forge.remove_forged(tool_name)
        except Exception as e:
            logger.error("Rollback removal failed for %s: %s", tool_name, e)

        # Restore old version if it existed
        if snapshot.code:
            try:
                from tain_agent.plugins.tool.forge_cycle import ImprovementSpec
                restore_spec = ImprovementSpec(
                    capability_id=f"restore_{tool_name}",
                    description=f"Restored version of {tool_name}",
                    function_name=tool_name,
                    parameters={},
                )
                self._tools.forge_cycle(restore_spec, code=snapshot.code)
                logger.info("Restored previous version of %s from snapshot", tool_name)
            except Exception as e:
                logger.error("Failed to restore snapshot for %s: %s", tool_name, e)

        if self._lineage:
            try:
                self._lineage.record_evolution(
                    None, None,
                    extra={"action": "rollback", "tool_name": tool_name,
                           "reason": "quality_degraded_or_verify_failed"},
                )
            except Exception:
                pass

    # ── Background Loop ───────────────────────────────────────────────────

    def _run_loop(self) -> None:
        """Background thread: periodically assess and execute evolution cycles."""
        logger.info("Evolution loop thread started")

        while self._running:
            try:
                if self._paused:
                    time_module.sleep(5)
                    continue

                if self._improvements_this_session >= self.max_improvements_per_session:
                    logger.info("Evolution quota exhausted (%d/%d)",
                                self._improvements_this_session,
                                self.max_improvements_per_session)
                    break

                # Rate limiting
                if self._last_cycle_at:
                    last = datetime.fromisoformat(self._last_cycle_at)
                    elapsed = (now() - last).total_seconds()
                    if elapsed < self.min_interval_seconds:
                        wait = self.min_interval_seconds - elapsed
                        time_module.sleep(min(wait, 30))
                        continue

                # Execute one cycle
                result = self.run_one_cycle()
                if result.skipped:
                    # Exponential backoff when no need
                    time_module.sleep(60)
                elif not result.success:
                    time_module.sleep(30)
                else:
                    time_module.sleep(5)

            except Exception as e:
                logger.exception("Evolution loop error: %s", e)
                time_module.sleep(30)

        logger.info("Evolution loop thread stopped")

    # ── State Persistence ─────────────────────────────────────────────────

    def _save_state(self) -> None:
        """Save loop state to disk for cross-session continuity."""
        try:
            state_dir = Path("agent_workspace/state")
            state_dir.mkdir(parents=True, exist_ok=True)
            state_file = state_dir / "evolution_loop.json"

            state = self.export_state()
            state["_saved_at"] = now().isoformat()

            with open(state_file, "w", encoding="utf-8") as f:
                json.dump(state, f, indent=2, ensure_ascii=False)
        except Exception:
            logger.debug("Failed to save evolution loop state", exc_info=True)

    def _load_state(self) -> None:
        """Load loop state from disk on startup."""
        try:
            state_file = Path("agent_workspace/state/evolution_loop.json")
            if not state_file.exists():
                return

            with open(state_file, "r", encoding="utf-8") as f:
                state = json.load(f)

            self._improvements_this_session = state.get("improvements_this_session", 0)
            self._last_cycle_at = state.get("last_cycle_at")
            self._last_trigger_scores = state.get("trigger_scores", {})

            # Expire state older than 24 hours
            if state.get("_saved_at"):
                saved_at = datetime.fromisoformat(state["_saved_at"])
                if (now() - saved_at).total_seconds() > 86400:
                    self._improvements_this_session = 0
                    self._last_cycle_at = None
        except Exception:
            logger.debug("Failed to load evolution loop state", exc_info=True)
